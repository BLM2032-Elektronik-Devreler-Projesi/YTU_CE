#include <stdio.h>
#define DIM 100
int main(void){
	int i, j, n, a[DIM], TMP;
	printf("please enter number of elements");
	scanf("%d", &n);
	for(j=0;j<n;j++){
	printf("please enter %d. element",j);
	scanf("%d", &a[j]);
	}
	i=1;
	j=n;
	while(i<j) {
		if(a[i]%2==0) {
		i+=1;
		}
		else {
		TMP=a[i];
		a[i]=a[j];
		a[j]=TMP;
		j=j-1;
		}
		}
		for(i=1;i<n;i++){
		printf("%d\n",a[i]);
		}
	
	
	
	
	
	
	return 0;
}
